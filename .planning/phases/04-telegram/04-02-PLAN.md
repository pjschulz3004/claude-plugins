export interface SchedulerConfig {
  yamlPath: string;

In the task execution flow (after dispatcher.dispatch returns and ledger records the result), add notification logic:

```typescript
// After recording result in ledger:
if (task.autonomy === "notify" && this.config.notifyChannels?.length) {
  const summary = result.subtype === "success"
    ? `Task "${task.name}" completed successfully.\n\n${result.result.slice(0, 500)}`
    : `Task "${task.name}" failed: ${result.result?.slice(0, 300) ?? "unknown error"}`;

  await sendNotification(
    this.config.notifyChannels,
    summary,
    { urgent: result.subtype !== "success" }  // failures are urgent (bypass quiet hours)
  );
}
```

Key behavior:
- Only tasks with `autonomy: "notify"` in heartbeat.yaml trigger notifications
- Task success = non-urgent (suppressed during quiet hours)
- Task failure = urgent (bypasses quiet hours -- Paul needs to know)
- Notification text includes task name and truncated result (first 500 chars for success, 300 for failure)
